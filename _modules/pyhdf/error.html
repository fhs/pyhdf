def _checkErr(procName, val, msg=""):

    if val is None or (not isinstance(val, str) and val < 0):
        #_C._HEprint();
        errCode = _C.HEvalue(1)
        if errCode != 0:
            err = "%s (%d): %s" % (procName, errCode, _C.HEstring(errCode))
        else:
            err = "%s : %s" % (procName, msg)
        raise HDF4Error(err)
